function sendSoloButtons()
  local allButtonGroups = root:findAllByName('SoloButtonGroup',true)
  local allButtons = {}

  for i = 1,#allButtonGroups do
    for j = 1,#allButtonGroups[i].children do
      allButtons[#allButtons + 1] = allButtonGroups[i].children[j]
    end
  end

  for i = 1, #allButtons do
    local button = allButtons[i]
    button:notify('callOSC')
  end

  --print(#allButtons)
end

===== [GROUP] Automatic Chaser Button Group =====
local TIMEOUT = 10000
local maxTimeout = 20000
local time
local activeButtonId
local loopIndex
local indexOfChaser

function init()
  time = getMillis()
  loopIndex = 1
  time = TIMEOUT + 1
  indexOfChaser = nil
  resetAllColors()
end

function update()

  TIMEOUT = root:findByName('Automatic Chaser Speed',true).values.x * maxTimeout
  if TIMEOUT < 1000 then
    TIMEOUT = 1000
  end

  local chasersStart = self.parent.children.StartStopAutoChasers.values.x == 0
  if chasersStart == false then
    resetAllColors()
  end

  if time and getMillis() - time > TIMEOUT and chasersStart == true then
    calculateActiveChaser()
  end
end

function calculateActiveChaser()
  resetAllColors()
  time = getMillis()
  local foundActiveButton = false
  local activeButtonValue = nil
  local noButtonPressed = true;
  local newIndexOfChaser = nil
  for i= 1,#self.children do

    if loopIndex > #self.children then
      loopIndex = 1
    end
    --print('loopIndex:'..loopIndex)


    local child = self.children[loopIndex]

    if child.type == ControlType.BUTTON and child.values.x == 1 then noButtonPressed = false end

    if indexOfChaser then
      if loopIndex == indexOfChaser and foundActiveButton == false then
        foundActiveButton = true
        activeButtonValue = child.values.x
        --print('foundactivebutton')
      elseif foundActiveButton and child.type == ControlType.BUTTON and child.values.x == 1 then
        --print('setting new button')
        newIndexOfChaser = loopIndex
        break
      end
    else
      --print('hre')
      if  child.type == ControlType.BUTTON and child.values.x == 1 then
        newIndexOfChaser = loopIndex
        break
      end
    end

    loopIndex = loopIndex + 1
  end

  if foundActiveButton and activeButtonValue == 1 and newIndexOfChaser == nil then
    newIndexOfChaser = indexOfChaser
  end

  if newIndexOfChaser ~= nil then
    self.children[newIndexOfChaser].color = Color(0.000000,0.899287,1.000000,1.000000)
  end

  if indexOfChaser ~= newIndexOfChaser then
    callPlayChaser(newIndexOfChaser)
  end

  indexOfChaser = newIndexOfChaser
  if indexOfChaser ~= nil then
    activeButtonId = self.children[indexOfChaser].ID
  else
    activeButtonId = nil
  end
  --print('end')

end

function resetAllColors()
  for i =  1,#self.children do
    if self.children[i].type == ControlType.BUTTON then
      self.children[i].color = Color(1.000000,0.098123,0.066667,1.000000)
    end
  end
end

function callPlayChaser(index)
  root:notify('AutoChaser',index)
  if index == nil then
    return
  end

  --print('chaser being called :'..index)
end

function onReceiveNotify(id,toggleValue)
  --print('notifyreceviedauto')
  --print(id)
  --print(toggleValue)
  --print(activeButtonId)
  if (toggleValue == 0 and activeButtonId == id) or (toggleValue == 1 and activeButtonId == nil) then
    print('aa')
    time = 0
    calculateActiveChaser()
  end
end

===== [GROUP] ChaserButtonTogglesGroup =====
function init()
  for i = 1,#self.children do
    local child = self.children[i]
    child.values.x = 0;
  end
end

function onReceiveNotify(key, value)
  print('notified!')
  local chaserToSend = nil;
  for i = 1,#self.children do
    local child = self.children[i]
    if child.ID ~= key then
      if child.values.x == 1 then
        chaserToSend = i
      end
    else
      if child.values.x == 0 then
        chaserToSend = nil
      else
        chaserToSend = i
      end
      break
    end
  end

  for i = 1,#self.children do
    local child = self.children[i]
    if i ~= chaserToSend then
      child.values.x = 0;
    end
  end

  sendChaserClicked(chaserToSend)
end

function sendChaserClicked(index)
  --print('sending chaser clicked""'..index)
  root:notify('ManualChaser',index)
end

function onReceiveNotify(key,input)
  local allMacroButtons = self:findAllByName('Macro Button')
  local allMacroButtonLabel = self:findAllByName('Macro Button Label')

  local bottomPages = root:findByProperty('tag','Master Controller Bottom Pages',true)
  bottomPages.values.page = 1
  local macroPickerLabel = self:findByProperty('tag','Light Setup Name Label',true)
  macroPickerLabel.values.text = input.lightName

  for i = 1,#allMacroButtons do
    local macroButton = allMacroButtons[i]
    macroButton.frame.h = 100
    macroButton.frame.w = 100
    macroButton.frame.x = macroButton.frame.w * ((i-1) % 5) + 70
    macroButton.frame.y = macroButton.frame.h * (math.floor((i-1)/5)) + 30
    macroButton.tag = ''
    if #input.macros > i - 1 then
      macroButton.visible = true
      macroButton.tag = input.lightName .. ' Macro ' ..input.macros[i].name
    else
      macroButton.visible = false
    end
  end

  for i = 1,#allMacroButtonLabel do
    local macroButtonLabel = allMacroButtonLabel[i]
    macroButtonLabel.frame.h = 100
    macroButtonLabel.frame.w = 100
    macroButtonLabel.frame.x = macroButtonLabel.frame.w * ((i-1) % 5) + 70
    macroButtonLabel.frame.y = macroButtonLabel.frame.h * (math.floor((i-1)/5)) + 30
    macroButtonLabel.values.text = ''
    macroButtonLabel.background = false
    if #input.macros > i - 1 then
      macroButtonLabel.visible = true
      macroButtonLabel.values.text = input.macros[i].name
    else
      macroButtonLabel.visible = false
    end
  end
end

local feedbackWaitMap = {}

function onReceiveNotify(from,values)
  if(from == 'onReceiveOSC') then
    onReceiveOSC(values.message,values.connections)
  end
  if(from == 'feedback') then
    print('asking for feedback')
    feedbackWaitMap[values.caller] = values.Id;
    sendOSC(values.caller,-1)
  end

end

-- example
function onReceiveOSC(message, connections)
  print('onReceiveOSC')
  local path = message[1]
  local arguments = message[2]
  print('\t path        =', path)
  for i=1,#arguments do
    print('\t argument    =', arguments[i].tag, arguments[i].value)
  end
  print('\t connections =', table.unpack(connections))

  if feedbackWaitMap[path] ~= nil then
    print('feedback returned!!')
    local caller = root:findByID(feedbackWaitMap[path],true)
    if caller ~= nil then
      caller:notify('feedback',arguments[1].value)
    end
  end
end
